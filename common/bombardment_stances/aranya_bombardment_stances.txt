hellraising = {
	default = no
	trigger= {
		owner = { has_technology = tech_aranya_hellraiser }
		owner = { has_ascension_perk = ap_aranya_galatic_depression }
		NOT = {
			owner = {
				has_policy_flag = orbital_bombardment_selective
			}
		}
	}

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 15
	army_damage = 100.0
	
	kill_pop_chance = {
		base = 0.2
	}

	min_pops_to_kill_pop = 0

	ai_weight = {
		weight = 0

		modifier = {
			factor = 10
			OR = {
				owner = { has_civic = civic_fanatic_purifiers }
				owner = { has_civic = civic_machine_terminator }
			}
		}
		# The devouring swarm will prefer eating population over hellraise their home.
		modifier = {
			factor = 2
			owner = { has_civic = civic_hive_devouring_swarm }
		}
		# xenophobe or spiritualist will most likly hellraise planet. 
		modifier = {
			factor = 10
			OR = {
				owner = { has_ethic = ethic_xenophobe }
				owner = { has_ethic = ethic_spiritualist }
			}
		}	
		# Fanatic xenophobe or spiritualist will often hellraise planet. 
		modifier = {
			factor = 30
			OR = {
				owner = { has_ethic = ethic_fanatic_xenophobe }
				owner = { has_ethic = ethic_fanatic_spiritualist }
			}
		}
		# If the fallen empire is xenophobe, will nuke the planet into the hellish world.
		modifier = {
			factor = 100
			owner = { has_ai_personality = awakened_fallen_empire_xenophobe }
		}
	}

}
