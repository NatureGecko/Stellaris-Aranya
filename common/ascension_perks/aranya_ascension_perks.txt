# Galactic Depression
ap_aranya_galatic_depression = {
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
	}

	on_enabled = {
		custom_tooltip = "ap_aranya_galatic_depression_effect"
		hidden_effect = { country_event = { id = AranyaSpecialProjectEvents.11001 } }
	}

	potential = {
		OR = {
			has_ethic = ethic_fanatic_xenophobe
			has_ethic = ethic_xenophobe
			has_authority = auth_machine_intelligence
		}
		NOR = {
			has_ascension_perk = ap_aranya_galatic_depression
			has_ascension_perk = ap_aranya_thumeran_ascendancy
		}
	}

	ai_weight = {
		factor = 1
		modifier = {
			factor = 15
			is_xenophobe = yes
			galaxy_percentage > 0.1
		}
		# As always
		modifier = {
			factor = 50
			OR = {
				has_civic = civic_hive_devouring_swarm
				has_civic = civic_machine_terminator
				has_civic = civic_fanatic_purifiers
				is_homicidal = yes
			}	
		}
		# If there are 3 empires with this perk, will no longer apply this perk to AI.
		modifier = {
			factor = 0
			count_playable_country = {
				count > 3
				limit = {
					has_ascension_perk = ap_aranya_galatic_depression
				}
			}
		}
	}

}

# Thumeran Ascendancy
ap_aranya_thumeran_ascendancy = {
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
	}

	potential = {
		NOT = {
			has_ascension_perk = ap_aranya_galatic_depression
		}
	}

	on_enabled = {
		custom_tooltip = "ap_aranya_thumeran_ascendancy_effect"
	}

	ai_weight = {
		factor = 1
		modifier = {
			factor = 1.3
			is_egalitarian = yes
		}
		modifier = {
			factor = 1.7
			has_ethic = ethic_materialist
		}
		modifier = {
			factor = 2.5
			has_ethic = ethic_fanatic_materialist
		}
		modifier = {
			factor = 0.2
			OR = {
				is_authoritarian = yes
				is_spiritualist = yes
			}
		}
	}
}
